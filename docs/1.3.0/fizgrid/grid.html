106    def add_event(
107        self,
108        time: int | float,
109        object,
110        method: str,
111        kwargs: dict = dict(),
112        priority: int = 0,
113    ) -> int:
114        """
115        Adds an event to the queue.
116        This method schedules an event for a specific object at a specific time.
117        Essentially, it allows you to schedule a method call on an object at a specific time.
118
119        Args:
120
121        - time (int|float): The time at which the event should occur.
122        - object: The object on which the event will occur.
123        - method (str): The name of the method to be called on the object.
124        - kwargs (dict): The keyword arguments to be passed to the method.
125        - priority (int): The priority of the event.
126            - Default: 0
127            - Higher values indicate higher priority.
128            - This is used to determine the order of events with the same time.
129            - If two events have the same time, the one with the higher priority will be processed first.
130            - If the priority is the same, the event with the lower ID will be processed first.
131            - In general:
132                - Adding an entity to the grid: 5
133                - Canceling a route: 4
134                - Handling collisions: 3
135                - Resolving an end of route: 2
136                - Removing an entity from the grid: 1
137                - Adding a Route: 0
138                - Adding an event: 0
139
140        Returns:
141
142        - int: The ID of the added event as generated by the queue.
143        """
144        return self.__queue__.add_event(
145            time=time,
146            event={"object": object, "method": method, "kwargs": kwargs},
147            priority=priority,
148        )

161    def resolve_next_state(self) -> list[dict]:
162        """
163        Resolves the next state of the grid.
164        This method processes the next event in the queue and updates the grid accordingly.
165
166        Returns:
167
168        - list[dict]: A list of events that were processed in this step.
169            - Each event is a dictionary containing a time, id and event.
170            - The event dictionary contains the object, method and kwargs.
171                - id (int): The ID of the event as generated by the queue.
172                - time (int|float): The time at which the event occurred.
173                - event (dict): The event that was processed.
174                    - object: The object on which the event occurred.
175                    - method (str): The name of the method that was called.
176                    - kwargs (dict): The keyword arguments that were passed to the method.
177        """
178        event_items = self.__queue__.get_next_events()
179        for event_item in event_items:
180            event = event_item.get("event")
181            object = event.get("object")
182            method = event.get("method")
183            kwargs = event.get("kwargs")
184            if object is None:
185                continue
186            getattr(object, method)(**kwargs)
187        return event_items

251        """
252        Runs the simulation for the grid.
